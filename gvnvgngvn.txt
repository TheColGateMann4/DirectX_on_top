
		template<DataType type>
		void AddColor(const char* elementName, ImGuiColorEditFlags flags = ImGuiColorEditFlags_None)
		{
			static_cast(type == Float3 || type == Float4, "using AddColor with non color elements");

			m_elements.push_back({ elementName, std::make_shared<LayoutElement>(DataType::Float, v_min, v_max, format, flags) });
		}

		void AddFloat(const char* elementName, float v_min = 0.0f, float v_max = 0.0f, const char* format = "%.3f", ImGuiSliderFlags flags = ImGuiSliderFlags_None)
		{
			m_elements.push_back({ elementName, std::make_shared<LayoutElement>(DataType::Float, v_min, v_max, format, flags) });
		}
